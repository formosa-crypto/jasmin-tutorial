fn memeq(reg u64 p q n) -> reg u64 {~~\loc{a00}~~
  reg u64 r one i;~~\loc{a01}~~
  r = 0;~~\loc{a02}~~
  one = 1;~~\loc{a03}~~
  i = 0;~~\loc{a04}~~
  while (i < n) {~~\loc{a05}~~
    if (r != 0) {~~\loc{a06}~~
      reg u64 a b;
      a = [p];~~\loc{a07}~~
      b = [q];~~\loc{a08}~~
      r = a != b ? one : r;~~\loc{a09}~~
      p += 8;~~\loc{a10}~~
      q += 8;~~\loc{a11}~~
    }~~\loc{a12}~~
    i = #INC(i);~~\loc{a13}~~
  }~~\loc{a14}~~
  return r;~~\loc{a15}~~
}~~\loc{a16}~~
